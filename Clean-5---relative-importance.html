Relative importance

a function that calculate, plot and save the relative importance for each y

boot.relimp() - conducts the bootstrap

booteval.relimp() - gets the confidence intervals

it also provide the relative in data fit to plot ternary plots

plot_my_rela <- function(model, response_name, plot_title = response_name,
                         save_plot = FALSE, upperl = 1, rela_plots = NULL) {
  
  # bootstrap relative importance
  
  rl_boot <- boot.relimp(model, b = 1000, rela = TRUE)
  ci <- booteval.relimp(rl_boot)
  
  print(ci)
  
  # data for bar plot
 
  data <- as.data.frame(ci@mark)
  colnames(data) <- c("var_explained", "sign", "lower_ci", "upper_ci")
  
  data <- data %>%
    mutate(
      var_explained = as.numeric(var_explained),
      lower_ci = as.numeric(lower_ci),
      upper_ci = as.numeric(upper_ci),
      variable = rownames(data),
      variable = case_when(
        grepl("tmean", variable) ~ "Temperature",
        grepl("protection", variable) ~ "Protection",
        grepl("Invasive_biomass", variable) ~ "NIS"
      ),
      variable = ordered(variable, c("Temperature", "Protection", "NIS"))
    )
  
  total_var <- ci@R2
  
  # data for ternary plot
  rela_out <- data %>%
    select(variable, var_explained) %>%
    pivot_wider(names_from = variable, values_from = var_explained) %>%
    mutate(response = response_name) %>%
    select(response, Temperature, Protection, NIS)
  
  # bar plot
  plot <- ggplot(data = data, aes(x = variable, y = var_explained, fill = variable)) +
    geom_bar(stat = "identity") +
    theme_classic() +
    ylab("Variance explained") +
    labs(
      title = plot_title,
      subtitle = bquote(R^2 == .(round((total_var * 100), 2)) * "%")
    ) +
    ylim(0, upperl) +
    theme(
      legend.position = "none",
      axis.title.x = element_blank(),
      axis.text = element_text(size = 20),
      axis.title = element_text(size = 24),
      plot.title = element_text(size = 26, hjust = 0.5),
      plot.subtitle = element_text(size = 18, hjust = 0.5)
    ) +
    scale_fill_manual(values = c("#364D8B", "#03718A", "#30A3D1"))
  
  
  if (save_plot) {
    plot_name <- paste0("Relative importance - ", plot_title, ".pdf")
    ggsave(
      plot = plot,
      filename = plot_name,
      width = 10,
      height = 8,
      dpi = 300,
      path = rela_plots
    )
  }
  
  return(list(
    plot = plot,
    rela = rela_out,
    ci_data = data,
    R2 = total_var
  ))
}

        axis.title = element_text(size = 9),
        legend.position = c(0.68, 0.88),

        plot.margin = margin(3, 3, 3, 3))

make_interaction_plot <- function(pred_data, raw_y, ylab_text, col_hex) {
  ggplot(data = pred_data) +
    geom_ribbon(aes(x = Temperature, ymin = conf.low, ymax = conf.high, fill = Protection), alpha = 0.2) +
    geom_line(aes(x = Temperature, y = predicted, color = Protection), linewidth = 0.7) +
    geom_point(data = diversity_table, aes(x = tmean, y = {{ raw_y }}, color = protection), alpha = 0.25, size = 0.8) +
    xlab("Temperature (°C)") +
    ylab(ylab_text) +
    scale_color_manual(values = c("grey60", col_hex)) +
    scale_fill_manual(values = c("grey60", col_hex)) +
    guides(fill = "none") +
    interaction_theme
}
